const { promisify } = require('util');
const sleep = promisify(setTimeout);

async function retryAsync(fn, options = {}) {
  const {
    retries = 0,
    minDelayMs = 0,
    factor = 1,
    jitter = 0,
    shouldRetry = () => true,
    signal = null,
  } = options;

  let attemptIndex = 0;
  while (true) {
    try {
      return await fn(attemptIndex);
    } catch (error) {
      attemptIndex++;
      if (attemptIndex > retries || !shouldRetry(error, attemptIndex)) {
        throw error;
      }
      if (signal?.aborted) {
        throw new Error('Operation aborted');
      }
      const delay = minDelayMs * Math.pow(factor, attemptIndex - 1);
      const jitterDelay = delay * (jitter * (Math.random() * 2 - 1));
      await sleep(delay + jitterDelay);
    }
  }
}

function topKFrequentWords(text, k, options = {}) {
  const { stopWords = [] } = options;
  const wordPattern = /\b[a-z0-9']+\b/gi;
  const wordCounts = new Map();

  text.replace(wordPattern, (word) => {
    const lowerWord = word.toLowerCase();
    if (!stopWords.includes(lowerWord)) {
      wordCounts.set(lowerWord, (wordCounts.get(lowerWord) || 0) + 1);
    }
  });

  const sortedWords = Array.from(wordCounts.entries())
    .sort((a, b) => {
      if (b[1] !== a[1]) {
        return b[1] - a[1];
      }
      return a[0].localeCompare(b[0]);
    })
    .slice(0, k);

  return sortedWords.map(([word, count]) => ({ word, count }));
}

module.exports = { retryAsync, topKFrequentWords };
